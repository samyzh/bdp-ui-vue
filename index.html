bdp-ui-vue组件库文档

基于 ant-design-vue 封装的组件库

快速入门 →

# 安装

yarn add @bdp/ui-vue -S

# Usage

import Vue from 'vue';
import { BModal } from '@bdp/ui-vue';
import App from './App';

Vue.component(BModal.name, BModal);

Vue.use(BModal);

new Vue({
    el: '#app',
    components: { App },
    template: '<App/>',
});
