# 封装优化

  • 封装 Icon 和 Button 组合的组件；
  • 设置图标类型和是否可用处理；

# 安装注册

import { BBatchButton } from '@bdp/ui-vue';
Vue.use(BBatchButton);

# 基本使用

# 代码演示

<template>
    <BBatchButton :menuList="menuList" @clickMenu="clickMenu"/> 
</template>

<script>
export default {
    data() {
        return {
            menuList: [
                {
                    value:'del',
                    label:'批量删除',
                    disabled:false,
                },
                {
                    value:'stop',
                    label:'批量停止',
                    disabled:true,
                },
            ]
        }
    },
    methods: {
        clickMenu (item, key, keyPath) {
            console.log(item, key, keyPath)
        },
    }
}
</script>
